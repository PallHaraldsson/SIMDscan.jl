AR(1)

T = 2500
ϵ = randn(T)
y = similar(ϵ)
α = 0.9
function ar1recursize!(y, ϵ, α)
    y[1] = ϵ[1]
    for t = 2:T
        y[t] = α*y[t-1] + ϵ[t]
    end
    y
end
ar(y,e) = (e[1] + α*y[1]*e[2], α*y[2]*e[2])
